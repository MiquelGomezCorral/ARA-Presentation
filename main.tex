\section{Introducción}

\begin{frame}{El problema: predecir sin tener todos los datos}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \vspace{0.3em}
      En la práctica clínica \hl{rara vez se dispone de todas las pruebas} de un paciente al momento de decidir.

      \medskip
      \textbf{¿Por qué faltan datos?}
      \begin{itemize}
        \item Urgencia: no hay tiempo para todas las secuencias MRI
        \item Fallos de equipamiento o captura
        \item Paciente inestable o no colaborador
        \item Coste económico de pruebas adicionales
        \item Registros incompletos o perdidos
      \end{itemize}

      \medskip
      \begin{block}{Objetivo}
        Predecir la esperanza de vida en glioblastoma con un modelo \hl{robusto a cualquier patrón de datos faltantes}, sin reentrenar.
      \end{block}
    \end{column}
    \begin{column}{0.42\textwidth}
      \centering
      \includegraphics[width=\textwidth]{images/UPENN-GBM-00465.png}
      \vspace{0.5em}

      {\small\textit{MRI 3D multimodal --- glioblastoma\\(UPenn-GBM)}}

      \vspace{0.8em}
      \begin{alertblock}{Pregunta clave}
        \small ¿Qué datos son \textit{realmente} imprescindibles para un buen diagnóstico?
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Dataset UPenn-GBM y tarea}
  \begin{columns}[T]

        IDH1 mutado / wildtype              & 19 / 514 \\
        MGMT metilado / no metilado         & 115 / 160 \\
        Resección total (GTR)               & 362 (57.5\%) \\

        \rowcolor{lightbg}

\section{Modalidades de entrada}

\begin{frame}{Tres modalidades, una arquitectura}
  \begin{columns}[T]
    \begin{column}{0.32\textwidth}
      \begin{block}{\centering Imagen MRI 3D}
        \centering
        \includegraphics[width=0.85\textwidth]{images/UPENN-GBM-00465.png}
        \vspace{0.3em}
        \small
        \begin{itemize}
          \item 4 canales: T1, T1Gd, T2, FLAIR
          \item Volumen reducido a\\$96^3$ vóxeles\\{\scriptsize (memoria / cómputo)}
          \item \textbf{Salida:} 217 tokens
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.32\textwidth}
      \begin{block}{\centering Datos clínicos}
        \vspace{0.3em}
        \small
        \begin{itemize}
          \item Género, edad, KPS
          \item IDH1, MGMT, GTR
          \item \textit{One-hot} + normalización $[-1, 1]$
          \item \textbf{14 \textit{features}}
          \item \textbf{Salida:} 8 tokens
        \end{itemize}
        \vspace{0.3em}
        {\scriptsize Falta $\to$ cero + \textit{bit flag} + \texttt{missing\_emb} aprendido}
      \end{block}
    \end{column}
    \begin{column}{0.32\textwidth}
      \begin{block}{\centering Radiómicos}
        \vspace{0.3em}
        \small
        \begin{itemize}
          \item 144 \textit{features} por\\(región, modalidad)
          \item 3 regiones: ED, ET, NC
          \item 4 modalidades MRI
          \item \textbf{Salida:} 12 tokens
        \end{itemize}
        \vspace{0.3em}
        {\scriptsize Complementan la imagen cuando faltan canales MRI}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.7em}
  \begin{center}
    \small\textit{El paper original descartaba los radiómicos por considerarlos ruido.\\Nosotros los incorporamos como modalidad complementaria gestionada por un \textbf{sigmoid gate}.}
  \end{center}
\end{frame}

\section{Arquitectura}

\begin{frame}{Arquitectura: visión general}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Fase 1 --- Pretraining auto-supervisado}
      \begin{itemize}
        \item Dos vistas aumentadas del mismo volumen
        \item \textit{Contrastive loss} (NT-Xent) + \textit{context reconstruction}
        \item Sobre BraTS (1\,251 vol.) o UPenn (603 vol.)
      \end{itemize}

      \medskip
      \textbf{Fase 2 --- Clasificación binaria}
      \begin{itemize}
        \item Encoder ViT congelado $\to$ \textit{tokens} de imagen
        \item Fusión con radiómicos vía \hl{sigmoid gate}
        \item \hl{Cross-attention} bidireccional\\imagen $\leftrightarrow$ clínicos
        \item \textit{Attention pooling} + clasificador binario
      \end{itemize}
    \end{column}
    \begin{column}{0.50\textwidth}
      \centering
      \fbox{\includegraphics[width=\textwidth]{images/architecture.png}}

      \vspace{0.5em}
      {\scriptsize\textit{Pipeline en dos fases: pretraining SSL (arriba) y predicción multimodal (abajo).}}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Cómo manejamos los datos faltantes}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{block}{Sigmoid gate (radiómicos $\leftrightarrow$ imagen)}
        \small
        \[
          \sigma = \mathrm{sigmoid}\!\bigl(\mathbf{W}\,[\,\text{img};\,\text{rad}\,]\bigr)
        \]
        \[
          \text{fused} = \text{img}\odot\sigma \;+\; \text{rad}\odot(1-\sigma)
        \]
        Decide \textbf{por token y por dimensión} cuánta información radiómica incorporar.

        \medskip
        Si los radiómicos faltan $\to$ \textbf{rad}$\approx 0$ $\to$ $\sigma\to 1$\\ $\to$ imagen pura, sin código condicional.
      \end{block}

      \medskip
      \begin{block}{\textit{Cross-attention} bidireccional}
        \small
        Cada modalidad se enriquece atendiendo a la otra.\\
        Pooling final + MLP de clasificación.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Sin ramas \texttt{if}: todo vía máscaras aprendidas}
        \small
        \begin{description}
          \item[\texttt{missing\_emb}] \textit{embedding} aprendido que sustituye \textit{features} tabulares ausentes
          \item[\texttt{image\_mask\_proj}] inyecta la presencia/ausencia de canales MRI directamente en el \textit{gate}
          \item[Máscaras de radiómicos] tokens inválidos $\to$ cero, ignorados por \textit{mean pool}
        \end{description}
      \end{block}

      \vspace{0.4em}
      \begin{exampleblock}{Principio de diseño}
        \small El modelo aprende \textbf{qué hacer cuando algo falta}, en vez de seguir reglas codificadas a mano.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Contribuciones}

\begin{frame}{Nuestras contribuciones sobre el paper base}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Robustez ante datos faltantes}
      \begin{itemize}
        \item \hl{Data dropout}: simulación sistemática de ausencias durante el entrenamiento (MRI, tabular, radiómicos)
        \item \hl{Mask wiring}: el modelo \textit{sabe qué falta}\\
              vía \texttt{missing\_emb} e \texttt{image\_mask\_proj}
        \item Manejo de ausencias \textbf{sin condicionales}: máscaras + \textit{gates} aprendidos
      \end{itemize}

      \medskip
      \textbf{Nueva modalidad}
      \begin{itemize}
        \item \hl{Radiómicos} como complemento (no ruido): el \textit{sigmoid gate} decide cuándo y cuánto usarlos
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Mejor representación de la imagen}
      \begin{itemize}
        \item \hl{Embeddings posicionales 3D}: rejilla $6^3$ aprendida que preserva la localidad espacial
        \item \hl{Pretraining con BraTS} (1\,251 vol.) para diversificar el SSL
        \item Aumentaciones de intensidad adaptadas al dominio BraTS
      \end{itemize}

      \medskip
      \textbf{Adaptación de la tarea}
      \begin{itemize}
        \item Clasificación binaria a 12 meses (en lugar de curva de supervivencia)
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.6em}
  \begin{center}
    \begin{exampleblock}{\centering Resultado}
      \centering\small
      El modelo generaliza a \textbf{cualquier patrón de datos faltantes} en inferencia, sin reentrenar.
    \end{exampleblock}
  \end{center}
\end{frame}

\begin{frame}{Diseño experimental}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Partición de datos (UPenn-GBM)}
      \medskip

      \centering
      \begin{tabular}{lrr}
        \toprule
        \textbf{Conjunto} & \textbf{Pacientes} & \textbf{\%} \\
        \midrule
        Entrenamiento & 482 & 80\% \\
        Validación    & 60  & 10\% \\
        Test          & 61  & 10\% \\
        \bottomrule
      \end{tabular}

      \medskip
      \raggedright
      \textbf{Simulación de datos faltantes}
      \begin{itemize}
        \item MRI: $\sim$5\% por canal
        \item Radiómicos: $\sim$10\% por región
        \item Tabulares: $\sim$10\% por grupo
      \end{itemize}

      \smallskip
      {\scriptsize \textit{Dynamic dropout} en \textit{train}; \textit{static dropout} fijo en test (reproducible).}
    \end{column}
    \begin{column}{0.50\textwidth}
      \textbf{Entrenamiento en dos fases}

      \medskip
      \textbf{1. Pretraining SSL}
      \begin{itemize}
        \item BraTS (1\,251 vol.) o UPenn (603 vol.)
        \item \textit{Batch} 32, AdamW $lr = 10^{-4}$, 20--50 épocas
        \item Aumentaciones: \textit{flip}, \textit{rotate}, \textit{noise}, \textit{crop}, \textit{patch swap}, \textit{cutout}
      \end{itemize}

      \medskip
      \textbf{2. Clasificación de supervivencia}
      \begin{itemize}
        \item ViT congelado
        \item \textit{Label smoothing} 0.1
        \item \textit{Cosine annealing} + \textit{early stopping}
        \item \textit{Batch} 16, AdamW $lr = 10^{-4}$
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Resultados}

\begin{frame}{Resultados --- Ablación de mejoras}
  \centering
  \small
  \begin{tabular}{clcccc}
    \toprule
    \textbf{Paso} & \textbf{Cambio introducido} & \textbf{F1} & \textbf{Prec.} & \textbf{Rec.} & $\boldsymbol{\Delta}$ \textbf{F1} \\
    \midrule
    0b & Baseline (UPenn SSL)                          & 0.6518 & 0.6657 & 0.6557 & --- \\
    1  & + \textit{Mask wiring}                        & 0.6382 & 0.6426 & 0.6393 & \bad{$-$0.0136} \\
    3  & + Normalización tabular $[-1, 1]$             & 0.6407 & 0.6926 & 0.6557 & \bad{$-$0.0111} \\

    \bottomrule
  \end{tabular}

  \vspace{0.7em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Mejora acumulada sobre el baseline}
        \centering\small
        \begin{tabular}{lcc}
          \toprule
          \textbf{Métrica} & \textbf{Baseline} & \textbf{Final} \\
          \midrule
          F1         & 0.6518 & \textbf{0.6640} \good{(+0.012)} \\
          Precisión  & 0.6657 & \textbf{0.6946} \good{(+0.029)} \\
          \textit{Recall} & 0.6557 & \textbf{0.6721} \good{(+0.016)} \\
          \bottomrule
        \end{tabular}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Observaciones clave}
        \small
        \begin{itemize}
          \item Los \textit{embeddings} 3D son la mejora individual de mayor impacto
          \item El \textit{mask wiring} requiere más épocas para converger
          \item El SSL con BraTS pierde estabilidad pasadas $\sim$20 épocas
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Resultados --- Efecto del \textit{data dropout}}
  \centering
  \small
  \begin{tabular}{cccc}
    \toprule
    \textbf{Dropout Train} & \textbf{Dropout Test} & \textbf{F1 (\%)} & \textbf{F1 con radiómicos (\%)} \\
    \midrule
    No  & No  & 63.38 & 58.55 \\
    Sí  & No  & 59.21 & 53.40 \\
    \rowcolor{lightbg}
    No  & Sí  & \textbf{68.47} & 59.56 \\
    Sí  & Sí  & 63.04 & 51.75 \\
    \bottomrule
  \end{tabular}

  \vspace{0.7em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Lecciones sobre dropout}
        \small
        \begin{itemize}
          \item \textit{Dynamic dropout} en \textit{train} introduce ruido excesivo y degrada el rendimiento
          \item Solo aplicarlo en test mejora el F1 hasta $\sim$68\%
          \item El \textit{batch size} 32 resulta crítico para el SSL contrastivo
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{El papel de los radiómicos}
        \small
        \begin{itemize}
          \item Con datos completos añaden algo de ruido al modelo
          \item Con datos faltantes aportan \textbf{señal complementaria} --- exactamente su rol diseñado
          \item La diferencia se reduce a mayor nivel de dropout
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.3em}
  {\scriptsize Nota: estos resultados corresponden a una versión previa del \textit{split} y no son directamente comparables con la ablación anterior.}
\end{frame}

\section{Discusión y conclusiones}

\begin{frame}{Discusión}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{\good{Fortalezas}}
      \begin{itemize}
        \item Fusión multimodal $>$ cualquier modalidad por separado
        \item El \textit{sigmoid gate} aprovecha los radiómicos sin degradar cuando faltan
        \item Los \textit{embeddings} 3D mejoran la representación espacial del tumor
        \item Aceptamos \textit{cualquier} combinación de entradas sin reentrenar
        \item El SSL sobre BraTS aporta diversidad de dominio
      \end{itemize}
    \end{column}
    \begin{column}{0.50\textwidth}
      \textbf{\bad{Debilidades}}
      \begin{itemize}
        \item El \textit{dynamic dropout} en \textit{train} es demasiado agresivo
        \item El SSL con BraTS colapsa pasadas $\sim$20 épocas (loss contrastiva)
        \item Dataset modesto (482 train): generalización limitada
        \item La clasificación binaria descarta granularidad temporal
        \item \textit{Mask wiring} converge lento --- necesita más épocas
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.7em}
  \begin{center}
    \begin{exampleblock}{Hallazgo principal}
      \centering\small
      No hacen falta ramas condicionales para manejar datos faltantes.\\
      \textbf{Máscaras aprendidas + \textit{sigmoid gate}} bastan para que la red\\decida qué información usar en cada caso.
    \end{exampleblock}
  \end{center}
\end{frame}

\begin{frame}{Conclusiones y trabajo futuro}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Conclusiones}
      \begin{enumerate}
        \item Arquitectura multimodal robusta que integra MRI, radiómicos y datos clínicos
        \item Datos faltantes gestionados \textbf{sin condicionales}: máscaras + \textit{gates} aprendidos
        \item Los radiómicos aportan señal complementaria cuando faltan canales MRI
        \item \textit{Embeddings} posicionales 3D y aumentaciones BraTS mejoran la base
        \item Mejora acumulada de \hl{+0.012 F1} sobre el baseline propio
      \end{enumerate}
    \end{column}
    \begin{column}{0.50\textwidth}
      \textbf{Trabajo futuro}
      \begin{enumerate}
        \item Pretraining SSL con más datasets (UCSF-PDGM, RHUH)
        \item \textit{Curriculum learning} sobre el dropout durante \textit{train}
        \item Predicción de curva de supervivencia (DeepHit) en lugar de binaria
        \item Validación con datos reales de centros con ausencias auténticas
        \item Análisis de qué modalidades son verdaderamente imprescindibles
        \item Destilar el modelo para despliegue clínico eficiente
      \end{enumerate}
    \end{column}
  \end{columns}
\end{frame}
